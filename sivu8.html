            <script>
              const nyt = new Date();
              const tunti = nyt.getHours();
              const bodyElement = document.body;
              const tervehdysElementti = document.getElementById('tervehdys');
    
              function setGreetingAndBackground() {
                  if (tunti >= 5 && tunti < 10) {
                tervehdysElementti.innerText = "Hyvää huomenta";
                bodyElement.style.backgroundImage = "url('kuva/morning.jpg')";
                  } else if (tunti >= 10 && tunti < 14) {
                tervehdysElementti.innerText = "Hyvää päivää";
                bodyElement.style.backgroundImage = "url('kuva/day.jpg')";
                  } else if (tunti >= 14 && tunti < 17) {
                tervehdysElementti.innerText = "Hyvää iltapäivää";
                bodyElement.style.backgroundImage = "url('kuva/evening.jpg')";
                  } else if (tunti >= 17 && tunti < 23) {
                tervehdysElementti.innerText = "Hyvää iltaa";
                bodyElement.style.backgroundImage = "url('kuva/night.jpg')";
                  } else {
                tervehdysElementti.innerText = "Hyvää yötä";
                bodyElement.style.backgroundImage = "url('kuva/yota.jpg')";
                  }
              }
    
              setGreetingAndBackground();
            </script>
