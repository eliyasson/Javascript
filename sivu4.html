        JS
          <script>
          const diceImages = [
              "kuva/dice1.png", 
              "kuva/dice2.png", 
              "kuva/dice3.png",
              "kuva/dice4.png", 
              "kuva/dice5.png", 
              "kuva/dice6.png"  
          ];
  
          function rollDice() {
              const dice = Math.floor(Math.random() * 6) + 1;
              displayDice(dice);
          }
  
          function displayDice(number) {
              const diceElement = document.getElementById('dice');
              const diceTextElement = document.getElementById('diceText');
  
              const imageFile = diceImages[number - 1];
              const text = `Arpa ${number}`;
  
              diceElement.src = imageFile;
              diceTextElement.innerText = text;
          }

        <div class="dice-container">
          <div class="dice-kuva">
            <button id="dice-button1" onclick="rollDice()">Heitä noppa</button>
          </div>
          <div class="diceContainer">
            <p id="diceText">Roll the dice</p>
            <img id="dice" src="kuva/dice.png" alt="Dice">     
          </div>
        </div>
