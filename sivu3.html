          <script>
            const box = document.getElementById('box');
    
            box.addEventListener('mousedown', function(event) {
                if (event.which === 1) {
                    box.innerText = 'Vasen painike';
                    box.style.backgroundColor = 'lightblue';
                } else if (event.which === 2) {
                    box.innerText = 'Keskimmäinen painike';
                    box.style.backgroundColor = 'lightgreen';
                } else if (event.which === 3) {
                    box.innerText = 'Oikea painike';
                    box.style.backgroundColor = 'lightcoral';
                }
            });
          </script>

Tehtävä 2 

        JS
          <script>
            const box1 = document.getElementById('box1');
    
            box1.addEventListener('mousedown', function(event) {
                if (event.which === 1) {
                    if (event.altKey) { 
                        const parallelConnectionResult = laskeRinnankytkentaTulos();
                        näytäTulos(parallelConnectionResult);
                    } else {
                        const seriesConnectionResult = laskeSarjakytkentaTulos();
                        näytäTulos(seriesConnectionResult);
                    }
                }
            });
    
            function laskeSarjakytkentaTulos() {
                const R1 = 10;
                const R2 = 20;
                const R3 = 30;
                const tulos = R1 + R2 + R3;
                return tulos;
            }
    
            function laskeRinnankytkentaTulos() {
                const R1 = 10;
                const R2 = 20;
                const R3 = 30;
                const tulos = 1 / ((1 / R1) + (1 / R2) + (1 / R3));
                return tulos;
            }
    
            function näytäTulos(tulos) {
                box1.innerText = "Tulos: " + tulos.toFixed(2) + " Ω";
            }
          </script>
         
        HTML-koodaus
          <div id="box1">Tulos näkyy täällä </div>
            Lue tehtävä klikkaamalla linkkiä...
Tulos näkyy täällä
