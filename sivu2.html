Tehtävä 2

Lyhyesti, mitä koodaus tekee. 

        JS
          <script>
          function areaLaske() {
            var radius = parseInt(document.getElementById("sade").value);
            var surfaceArea = 4 * Math.PI * (radius * radius);
            document.getElementById("areaResult").innerHTML = "The Area of sphere is: " + surfaceArea.toFixed(2);
          }
          function volumeLaske() {
            var radius = parseInt(document.getElementById("sade").value);
            var volume = 4/3 * Math.PI * (radius * radius * radius);
            document.getElementById("areaResult").innerHTML = "The volume of sphere is: " + volume.toFixed(2);
          }
          </script>
         
        HTML-koodaus
          <div>
            <h1 style="color: #333; font-family: arial, sans-sarif;">Pallon laskenta</h1>
            <p>Syötä pallon säde:</p>
            <input id="sade" type="text" placeholder="Anna pallon säde: ">
            <button id="area" onclick="areaLaske()">Laske/<button>
            <button id="volueme" onclick="volumeLaske()">Laske</button>
    
            <p id="areaResult"></p>
            <p id="volumeResult"></p>
    
          </div>
