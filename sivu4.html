        JS
          <script>
            function rollDice() {
                const dice = Math.floor(Math.random() * 6) + 1;
                document.getElementById('diceText').innerHTML = "Arpa " + dice;
    
                if (dice === 1) {
                  document.getElementById("dice").src = "kuva/dice1.png"
                } else if (dice ===2) {
                  document.getElementById("dice").src = "kuva/dice2.png"
                } else if (dice ===3) {
                  document.getElementById("dice").src = "kuva/dice3.png"
                } else if (dice ===4) {
                  document.getElementById("dice").src = "kuva/dice4.png"
                } else if (dice ===5) {
                  document.getElementById("dice").src = "kuva/dice5.png"
                } else {
                  document.getElementById("dice").src = "kuva/dice6.png"
                }
            }
          </script>
